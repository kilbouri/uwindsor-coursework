	\noindent
	1. Simplify the following using 3-variable K-Maps:
	\begin{multicols}{2}
		\begin{indented}{5mm}
			a. $F(x, y, z) = \sum m_{0, 2, 4, 5}$
			\begin{indented}{5mm}
				\begin{karnaugh-map}[4][2][1][yz][x]
					\minterms{0, 2, 4, 5}
					\maxterms{1, 3, 6, 7}

					% implicant groups
					\implicant{4}{5}
					\implicantedge{0}{0}{2}{2}
				\end{karnaugh-map}

				$\therefore F = z'x' + y'x$
			\end{indented}

			c. $F(x, y, z) = \sum m_{0, 1, 2, 3, 5}$
			\begin{indented}{5mm}
				\begin{karnaugh-map}[4][2][1][yz][x]
					\minterms{0, 1, 2, 3, 5}
					\maxterms{4, 6, 7}
	
					% implicant groups
					\implicant{1}{5}
					\implicant{0}{2}
				\end{karnaugh-map}
	
				$\therefore F = x' + y'z$
			\end{indented}
		\end{indented}
		\columnbreak
		\begin{indented}{5mm}
			\pagebreak
			b. $F(x, y, z) = \sum m_{0, 2, 4, 5, 6}$
